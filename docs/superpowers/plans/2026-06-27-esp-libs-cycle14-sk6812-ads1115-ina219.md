# esp-libs cycle 14 — SK6812 + ADS1115 + INA219 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add three independent L3 drivers to esp-libs — `sk6812` (RGBW LED), `ads1115` (16-bit 4-ch I2C ADC), `ina219` (I2C current/power monitor) — each a host-tested pure part + SDK glue.

**Architecture:** Three header-split L3 drivers: pure `<lib>.{h,c}` (host-Unity-tested) + SDK glue `<lib>_dev.{h,c}` (xtensa COMPILE_GATE deferred). `sk6812` mirrors `ws2812` (GPIO bit-bang); `ads1115`/`ina219` use the existing `i2c_bus` register API (16-bit big-endian registers). No bus prerequisite. Task 4 wires the build.

**Tech Stack:** C11, host `cc` + Unity (`../common/third_party/unity`), GNU make; ESP8266 RTOS SDK for the glue (deferred).

**Spec:** `docs/superpowers/specs/2026-06-27-esp-libs-cycle14-sk6812-ads1115-ina219-design.md`

## Global Constraints

- **Branch `feat/esp-libs-cycle14-afe`** (`git checkout -b feat/esp-libs-cycle14-afe` before Task 1).
- **Commands run from `clibs/esp-libs/`.** Unity at `../common/third_party/unity`.
- **Naming:** pure part = `<lib>.{h,c}` (freestanding, named in `test_<lib>.c`); glue = `<lib>_dev.{h,c}` (includes SDK headers).
- **Pure files** freestanding (`<stdint.h>`/`<stddef.h>`), no SDK, no malloc, host-testable.
- **Glue files** SDK (`esp_err.h`, `i2c_bus.h`, or `driver/gpio.h`/`rom/ets_sys.h`) → **not host-runnable; COMPILE_GATE deferred** (xtensa absent).
- **No UB:** ADS1115/INA219 conversions widen to `int64` before the multiply; shifts on positive uint8/uint16/uint32 only; no left-shift of a possibly-negative value.
- **Guards** `CLIBS_<MODULE>_H`; doc comment per public fn; NULL-guard every pointer param; "Not reentrant".
- **Units:** ADS1115 → microvolts; INA219 → bus mV / shunt µV / current µA / power µW.
- **Flags:** host `-std=c11 -Wall -Wextra -g`; strict `-std=c11 -Wall -Wextra -Werror`; sanitize `-std=c11 -fsanitize=undefined,address -fno-sanitize-recover=all` (`ASAN_OPTIONS=detect_leaks=0` on darwin).

```c
#include "sk6812.h"

/* Scale a channel by brightness/255. uint16 product <= 255*255 = 65025: no overflow, no UB. */
static uint8_t scale(uint8_t c, uint8_t brightness) {
    return (uint8_t) (((uint16_t) c * (uint16_t) brightness) / 255u);
}

size_t sk6812_render(const sk6812_rgbw *px, size_t n, uint8_t brightness, uint8_t *out, size_t out_size) {
    if (!px || !out || n == 0) {
        return 0;
    }
    size_t need = n * 4u;
    if (out_size < need) {
        return 0;
    }
    for (size_t i = 0; i < n; i++) {
        out[i * 4 + 0] = scale(px[i].g, brightness);   /* G */
        out[i * 4 + 1] = scale(px[i].r, brightness);   /* R */
        out[i * 4 + 2] = scale(px[i].b, brightness);   /* B */
        out[i * 4 + 3] = scale(px[i].w, brightness);   /* W */
    }
    return need;
}
```

`sk6812_dev.c` (not host-compiled):
```c
#include "sk6812_dev.h"
#include "rom/ets_sys.h"   /* ets_delay_us */

/* SK6812 @800kHz, 32 bits/pixel (G,R,B,W). ets_delay_us is too coarse for the sub-us
 * highs — FIRST CUT structure; on hardware replace per-bit highs/lows with a
 * cycle-counted (xthal_get_ccount) routine, interrupts off. Verified only by the
 * deferred esp-gate link. */
esp_err_t sk6812_init(sk6812 *self, gpio_num_t pin, uint16_t count, uint8_t *grbw_buf, size_t grbw_size) {
    if (!self || !grbw_buf) {
        return ESP_ERR_INVALID_ARG;
    }
    if (grbw_size < (size_t) count * 4u) {
        return ESP_ERR_INVALID_SIZE;
    }
    self->pin = pin;
    self->count = count;
    self->grbw = grbw_buf;
    self->grbw_size = grbw_size;
    gpio_set_direction(pin, GPIO_MODE_OUTPUT);
    gpio_set_level(pin, 0);
    return ESP_OK;
}

static void sk6812_send_byte(gpio_num_t pin, uint8_t b) {
    for (int i = 7; i >= 0; i--) {        /* MSB first */
        if (b & (1u << i)) {
            gpio_set_level(pin, 1); ets_delay_us(1); gpio_set_level(pin, 0);
        } else {
            gpio_set_level(pin, 1); gpio_set_level(pin, 0); ets_delay_us(1);
        }
    }
}

esp_err_t sk6812_show(sk6812 *self, const sk6812_rgbw *px, uint8_t brightness) {
    if (!self || !px) {
        return ESP_ERR_INVALID_ARG;
    }
    size_t n = sk6812_render(px, self->count, brightness, self->grbw, self->grbw_size);
    if (n == 0) {
        return ESP_ERR_INVALID_SIZE;
    }
    for (size_t i = 0; i < n; i++) {
        sk6812_send_byte(self->pin, self->grbw[i]);
    }
    ets_delay_us(80);                     /* latch/reset (>=80us for SK6812) */
    return ESP_OK;
}
```

```c
#ifndef CLIBS_ADS1115_H
#define CLIBS_ADS1115_H

#include <stdint.h>

/* Programmable-gain full-scale range (config PGA field [11:9] = 0..5). */
typedef enum {
    ADS1115_PGA_6V144 = 0, ADS1115_PGA_4V096, ADS1115_PGA_2V048,
    ADS1115_PGA_1V024, ADS1115_PGA_0V512, ADS1115_PGA_0V256
} ads1115_pga;

/* Signed 16-bit conversion code -> microvolts for the given PGA. uV = raw * FSR_uV / 32768
 * (int64 intermediate — raw*FSR_uV exceeds int32). pga out of range -> ADS1115_PGA_2V048. */
int32_t ads1115_to_microvolts(int16_t raw, ads1115_pga pga);

/* Build the 16-bit config register for a single-shot, single-ended read of `channel`
 * (0..3, masked &3) at `pga`: OS=1, MUX=0b100+channel, MODE=1, DR=0b100 (128 SPS),
 * comparator disabled. */
uint16_t ads1115_config(uint8_t channel, ads1115_pga pga);

#endif /* CLIBS_ADS1115_H */
```

```c
#include "ads1115.h"

/* Full-scale range in microvolts, indexed by ads1115_pga. */
static const int32_t ADS1115_FSR_UV[6] = { 6144000, 4096000, 2048000, 1024000, 512000, 256000 };

int32_t ads1115_to_microvolts(int16_t raw, ads1115_pga pga) {
    int idx = (pga <= ADS1115_PGA_0V256) ? (int) pga : (int) ADS1115_PGA_2V048;
    /* widen to int64 BEFORE the multiply: raw * FSR_uV exceeds int32. */
    return (int32_t) ((int64_t) raw * ADS1115_FSR_UV[idx] / 32768);
}

uint16_t ads1115_config(uint8_t channel, ads1115_pga pga) {
    uint16_t mux = (uint16_t) (0x4u | (channel & 0x3u));          /* 0b100 + channel: AINx vs GND */
    uint16_t pgabits = (uint16_t) ((pga <= ADS1115_PGA_0V256) ? pga : ADS1115_PGA_2V048);
    uint16_t cfg = 0x8000u                  /* OS = 1 (start single conversion) */
                 | (uint16_t) (mux << 12)
                 | (uint16_t) (pgabits << 9)
                 | 0x0100u                  /* MODE = 1 (single-shot) */
                 | 0x0080u                  /* DR = 0b100 (128 SPS) */
                 | 0x0003u;                 /* COMP_QUE = 0b11 (comparator disabled) */
    return cfg;
}
```

```c
#include "ads1115_dev.h"
#include "i2c_bus.h"
#include "rom/ets_sys.h"   /* ets_delay_us */

#define ADS1115_REG_CONV   0x00
#define ADS1115_REG_CONFIG 0x01

esp_err_t ads1115_init(ads1115 *self, uint8_t addr, ads1115_pga pga) {
    if (!self) {
        return ESP_ERR_INVALID_ARG;
    }
    self->addr = addr;
    self->pga = pga;
    return ESP_OK;
}

esp_err_t ads1115_read(ads1115 *self, uint8_t channel, int32_t *microvolts) {
    if (!self || !microvolts) {
        return ESP_ERR_INVALID_ARG;
    }
    uint16_t cfg = ads1115_config(channel, self->pga);
    uint8_t w[2] = { (uint8_t) (cfg >> 8), (uint8_t) (cfg & 0xFFu) };   /* big-endian */
    esp_err_t err = i2c_bus_write_reg(self->addr, ADS1115_REG_CONFIG, w, 2);
    if (err != ESP_OK) {
        return err;
    }
    ets_delay_us(8000);   /* 128 SPS -> ~7.8 ms conversion (first cut) */
    uint8_t r[2] = { 0 };
    err = i2c_bus_read_reg(self->addr, ADS1115_REG_CONV, r, 2);
    if (err != ESP_OK) {
        return err;
    }
    int16_t raw = (int16_t) (((uint16_t) r[0] << 8) | r[1]);
    *microvolts = ads1115_to_microvolts(raw, self->pga);
    return ESP_OK;
}
```

```c
#ifndef CLIBS_INA219_H
#define CLIBS_INA219_H

#include <stdint.h>

/* Bus-voltage register (0x02): bits 15..3 in 4 mV LSB -> millivolts. */
int32_t ina219_bus_mv(uint16_t raw);

/* Shunt-voltage register (0x01): signed, 10 uV LSB -> microvolts. */
int32_t ina219_shunt_uv(int16_t raw);
```

```c
#include "ina219_dev.h"
#include "i2c_bus.h"

#define INA219_REG_CONFIG  0x00
#define INA219_REG_SHUNT   0x01
#define INA219_REG_BUS     0x02
#define INA219_REG_POWER   0x03
#define INA219_REG_CURRENT 0x04
#define INA219_REG_CALIB   0x05
#define INA219_CONFIG_DEFAULT 0x399F   /* 32V, +/-320mV PGA, 12-bit, continuous shunt+bus */

static esp_err_t ina219_write16(uint8_t addr, uint8_t reg, uint16_t val) {
    uint8_t w[2] = { (uint8_t) (val >> 8), (uint8_t) (val & 0xFFu) };   /* big-endian */
    return i2c_bus_write_reg(addr, reg, w, 2);
}

static esp_err_t ina219_read16(uint8_t addr, uint8_t reg, uint16_t *out) {
    uint8_t r[2] = { 0 };
    esp_err_t err = i2c_bus_read_reg(addr, reg, r, 2);
    if (err != ESP_OK) {
        return err;
    }
    *out = (uint16_t) (((uint16_t) r[0] << 8) | r[1]);
    return ESP_OK;
}

esp_err_t ina219_init(ina219 *self, uint8_t addr, uint32_t shunt_milliohm, uint32_t max_current_ma) {
    if (!self) {
        return ESP_ERR_INVALID_ARG;
    }
    self->addr = addr;
    self->current_lsb_ua = max_current_ma * 1000u / 32768u;
    esp_err_t err = ina219_write16(addr, INA219_REG_CONFIG, INA219_CONFIG_DEFAULT);
    if (err != ESP_OK) {
        return err;
    }
    return ina219_write16(addr, INA219_REG_CALIB, ina219_calibration(self->current_lsb_ua, shunt_milliohm));
}

esp_err_t ina219_read_bus_mv(ina219 *self, int32_t *bus_mv) {
    if (!self || !bus_mv) {
        return ESP_ERR_INVALID_ARG;
    }
    uint16_t raw = 0;
    esp_err_t err = ina219_read16(self->addr, INA219_REG_BUS, &raw);
    if (err != ESP_OK) {
        return err;
    }
    *bus_mv = ina219_bus_mv(raw);
    return ESP_OK;
}

esp_err_t ina219_read_shunt_uv(ina219 *self, int32_t *shunt_uv) {
    if (!self || !shunt_uv) {
        return ESP_ERR_INVALID_ARG;
    }
    uint16_t raw = 0;
    esp_err_t err = ina219_read16(self->addr, INA219_REG_SHUNT, &raw);
    if (err != ESP_OK) {
        return err;
    }
    *shunt_uv = ina219_shunt_uv((int16_t) raw);
    return ESP_OK;
}

esp_err_t ina219_read_current_ua(ina219 *self, int32_t *current_ua) {
    if (!self || !current_ua) {
        return ESP_ERR_INVALID_ARG;
    }
    uint16_t raw = 0;
    esp_err_t err = ina219_read16(self->addr, INA219_REG_CURRENT, &raw);
    if (err != ESP_OK) {
        return err;
    }
    *current_ua = ina219_current_ua((int16_t) raw, self->current_lsb_ua);
    return ESP_OK;
}

esp_err_t ina219_read_power_uw(ina219 *self, int32_t *power_uw) {
    if (!self || !power_uw) {
        return ESP_ERR_INVALID_ARG;

    uint16_t raw = 0;

    if (err != ESP_OK) {
        return err;
    }
    *power_uw = ina219_power_uw(raw, self->current_lsb_ua);
    return ESP_OK;
}
```
